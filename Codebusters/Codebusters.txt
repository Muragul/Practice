rules

h1{
color: blue;

animation-timing-function:
linear; - равномерно
ease; - медленно-разгоняется-замедляется
ease-out; - быстро - замедление
ease-in; - медленно - ускорение
ease-in-out; - медленно начинается и заканчивается
step-start; - steps(1, start) = разбивает анимацию на отрезки, изменения происходят в начале шага 
step-end; - steps(1, end) = пошаговая анимация, изменения происходят в конце каждого шага 
steps( кол-во шагов, положение шага) = положение шага
       jump-start; jump-end; jump-none; jump-both

animation-name: list of animations; 
none; - отсутствие анимации
initial; - значение свойства по умолчанию
inherit; - наследует значение свойства от родительского элемента

animation-duration: s/ms;
animation-duration: 15s, 30s, 200ms;

animation-iteration-count: количество циклов
infinite;
real number;

animation-direction:
normal; = default
reverse; = все повторы в обратном направлении
alternate; = четные повторы в обратном направлении
alternate-reverse; = нечетные повторы в обратном направлении
initial;
inherit;
количество повторов начинается с 1

animation-play-state
running = анимация выполняется
paused = приостановлена
initial
inherit
состояние :hover = останавливает анимацию при наведении курсора мыши на объект

animation-delay
определяет когда анимация начнется
s/ms

animation-fill-mode
состояние элемента до и после воспроизведения анимации
none; =default
forwards; = останется как в конце
backwards; = вернется к начальному кадру
both; = первый ключевой кадр до начала и последний в конце
